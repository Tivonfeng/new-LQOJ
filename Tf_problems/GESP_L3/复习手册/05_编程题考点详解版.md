### 1️⃣ 数组与标记法（**最高频考点**）

**题型特点**：使用数组记录状态、进行统计或替换操作

| 题目 | 核心操作 | 关键技巧 |
|-----|---------|---------|
| 春游 | 布尔数组标记到达状态 | 初始化false，遍历标记 |
| 小杨的储蓄 | 数组累加 | 第i天存i元，索引对应 |
| 数字替换 | 两遍扫描 | 先找max/min，再替换 |
| 智慧购物 | 多组数据统计最小值 | 初始化大值，边读边更新 |
| 数组清零 | 循环模拟操作 | 找最大值下标、非0最小值 |

**模板代码**：

```cpp
// 标记法模板
bool visited[1005] = {false};  // 使用静态数组
for (int i = 0; i < m; i++) {
    visited[id[i]] = true;  // 标记到达
}
// 检查未标记的位置
for (int i = 0; i < n; i++) {
    if (!visited[i]) cout << i << " ";
}
```

```cpp
// 两遍扫描模板

for (int i = 1; i < n; i++) {
    if (a[i] > maxVal) maxVal = a[i];
    if (a[i] < minVal) minVal = a[i];
}
for (int i = 0; i < n; i++) {
    if (a[i] > k) a[i] = maxVal;
    else if (a[i] < k) a[i] = minVal;
}
```

```cpp
// 多组数据最小值统计
int minPrice[100005];  // 根据数据范围定义

    minPrice[i] = 1000000000;  // 初始化为较大值
}

    int kind, price;
    cin >> kind >> price;
    if (price < minPrice[kind]) minPrice[kind] = price;

for (int i = 0; i < M; i++) {
    total += minPrice[i];

```

**易错点**：
- 数组初始化：布尔数组用`{false}`或循环初始化
- 数值数组初始化为0或一个足够大/小的值
- 索引范围：从0还是从1开始，注意边界

### 4️⃣ 数学思维与逻辑推理（**核心考点**）

**题型特点**：建立数学模型，推导求解

| 题目 | 核心思想 | 关键公式 |
|-----|---------|---------|
| 小猫分鱼 | 逆向推导 | (鱼数 - i) × N / (N-1) |
| 完全平方数 | 双层遍历 | sqrt检查或平方比较 |
| 平衡序列 | 前缀和 | 前缀和×2 = 总和 |
| 分糖果 | 贪心递增 | candy[i] = max(a[i], candy[i-1]+1) |
| 寻找倍数 | 最大值性质 | 最大值是其他所有数的倍数 |

**模板代码**：

```cpp
// 小猫分鱼 - 逆向推导
int fish = 0;
while (true) {
    int temp = fish;
    bool ok = true;
    for (int j = 0; j < N; j++) {
        if ((temp - i) % (N - 1) != 0) {
            ok = false;
            break;
        }
        temp = (temp - i) / (N - 1) * N;
    }
    if (ok && temp > 0) {
        cout << fish << endl;
        break;
    }
    fish++;
}
```

```cpp
// 平衡序列 - 前缀和

}

for (int i = 0; i < n - 1; i++) {
    prefix += a[i];
    if (prefix * 2 == total) {
        cout << "Yes" << endl;
        return 0;
    }
}
cout << "No" << endl;
```

```cpp

}

```

```cpp
bool checkPassword(char s[]) {
    int len = 0;
    while (s[len] != '\0') len++;
    
    if (len < 6 || len > 12) return false;
    
    int hasUpper = 0, hasLower = 0, hasDigit = 0, hasSpecial = 0;
    for (int i = 0; i < len; i++) {
        char c = s[i];
        if (c >= 'A' && c <= 'Z') hasUpper = 1;
        else if (c >= 'a' && c <= 'z') hasLower = 1;
        else if (c >= '0' && c <= '9') hasDigit = 1;
        else if (c == '!' || c == '@' || c == '#' || c == '$') hasSpecial = 1;
    }
    
    int typeCount = hasUpper + hasLower + hasDigit + hasSpecial;
    return typeCount >= 2 && hasSpecial >= 1;
}
```

```cpp
int findMinFish(int N, int i) {
    int fish = 1;
    while (true) {
        int temp = fish;
        bool ok = true;
        for (int j = 0; j < N; j++) {
            if ((temp - i) % (N - 1) != 0 || (temp - i) < 0) {
                ok = false;
                break;
            }
            temp = (temp - i) / (N - 1) * N;
        }
        if (ok) return fish;
        fish++;
    }
}
```

**2. 前缀和**

```cpp
bool isBalanced(int a[], int n) {
    long long total = 0;
    for (int i = 0; i < n; i++) total += a[i];
    
    long long prefix = 0;
    for (int i = 0; i < n - 1; i++) {
        prefix += a[i];
        if (prefix * 2 == total) return true;
    }
    return false;
}
```

**3. 贪心策略**

```cpp
long long minCandies(int a[], int n) {
    long long candy[1005];
    candy[0] = a[0];
    for (int i = 1; i < n; i++) {
        candy[i] = max((long long)a[i], candy[i-1] + 1);
    }
    long long sum = 0;
    for (int i = 0; i < n; i++) sum += candy[i];
    return sum;
}
```

```cpp
void printDigits(char s[]) {
    // 定义4个数字的5行图案
    char pattern[4][5][6] = {
        {"****.", "****.", ".....", "****.", "....."},  // 0
        {"..*..", ".**..", "..*..", "..*..", ".***."},  // 1
        {".***.", "*...*", "....*", ".**..", "*****"},   // 2
        {".***.", "....*", "..**.", "....*", ".***."}    // 3
    };
    
    // 按行收集输出
    for (int row = 0; row < 5; row++) {
        for (int i = 0; s[i] != '\0'; i++) {
            int d = s[i] - '0';
            cout << pattern[d][row];
        }
        cout << endl;
    }
}
```

**2. 日历制作**

```cpp
void printCalendar(int month) {
    int days[13] = {0, 31, 28, 31, 30, 31, 30, 31, 31, 30, 31, 30, 31};
    cout << "MON TUE WED THU FRI SAT SUN" << endl;
    
    int startDay = 0;  // 2025年9月1日是周一
    for (int m = 1; m < month; m++) {
        startDay = (startDay + days[m]) % 7;
    }
    
    // 打印前面空格
    for (int i = 0; i < startDay; i++) {
        cout << "    ";
    }
    
    // 打印日期
    for (int d = 1; d <= days[month]; d++) {
        if (d < 10) cout << " " << d << "  ";
        else cout << d << "  ";
        
        if ((startDay + d) % 7 == 0) cout << endl;
    }
}
```
