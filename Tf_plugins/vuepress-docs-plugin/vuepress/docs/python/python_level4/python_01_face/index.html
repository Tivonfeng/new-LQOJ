from lqcodeAI import lq
import streamlit as st

st.title('我的颜值分析小助手')
st.write('📸 上传你的照片，AI帮你分析颜值！')

image = st.file_uploader('请上传一张照片', type=['jpg', 'png'])
if st.button('✨ 开始分析'):
    result = lq.face_analysis(password='lqcode', image=image)
    st.success('分析完成！')
    st.write(f'颜值分数：{result["beauty_score"]}')
    st.write(f'预测年龄：{result["age"]}')
    st.write(f'颜值建议：{result["suggestions"]}')
    st.balloons()
