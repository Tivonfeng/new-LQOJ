🧰 3.4 学习数据处理与展示技巧

🎨 数据美化与交互：让你的应用更专业 🎨

import streamlit as st
import pandas as pd
from datetime import datetime

# 显示当前时间
now = datetime.now().strftime("%Y-%m-%d %H:%M:%S")
st.write(f"数据更新时间：{now}")

# 高亮显示重要数据
def highlight_max(s):
    is_max = s == s.max()
    return ['background-color: yellow' if v else '' for v in is_max]

# 应用样式到数据表
st.dataframe(df.style.apply(highlight_max))

# 添加下载按钮
@st.cache_data
def convert_df(df):
    return df.to_csv().encode('utf-8')

csv = convert_df(df)
st.download_button(
    "下载数据",
    csv,
    "热搜榜.csv",
    "text/csv",
    key='download-csv'
)

from lqcodeAI import lq
import streamlit as st
import pandas as pd
from datetime import datetime

st.title('今日头条热搜榜')
st.write('📱 实时查看最热门的话题！')

# 显示当前时间
now = datetime.now().strftime("%Y-%m-%d %H:%M:%S")
st.write(f"数据更新时间：{now}")

# 添加刷新按钮
if 'hot_news' not in st.session_state or st.button('🔄 刷新数据'):
    with st.spinner('数据获取中...'):
        hot_news = lq.ai_hotnews(password='lqcode')
        st.session_state.hot_news = hot_news
    st.success('数据更新成功！')

# 显示热搜数据
if 'hot_news' in st.session_state:
    hot_news = st.session_state.hot_news
    
    # 转换为DataFrame
    data = []
    for item in hot_news:
        data.append({
            "排名": item["rank"],
            "话题": item["title"],
            "热度": item["hot_value"]
        })
    df = pd.DataFrame(data)
    
    # 创建标签页
    tab1, tab2 = st.tabs(["📊 数据表", "📈 图表"])
    
    with tab1:
        # 显示数据表
        st.dataframe(df, use_container_width=True)
        
        # 添加下载按钮
        @st.cache_data
        def convert_df(df):
            return df.to_csv().encode('utf-8')

        csv = convert_df(df)
        st.download_button(
            "📥 下载热搜数据",
            csv,
            "热搜榜.csv",
            "text/csv",
            key='download-csv'
        )
    
    with tab2:
        # 显示条形图
        st.subheader("热搜话题热度排行")
        chart_data = df.set_index("话题")
        st.bar_chart(chart_data["热度"])
        
        # 添加热门话题分析
        st.subheader("🔥 热门话题分析")
        top_topic = df.iloc[0]["话题"]
        st.write(f"当前最热门的话题是：**{top_topic}**")
        
        # 计算平均热度
        avg_hot = df["热度"].mean()
        st.write(f"当前热搜平均热度：**{avg_hot:.2f}**")
