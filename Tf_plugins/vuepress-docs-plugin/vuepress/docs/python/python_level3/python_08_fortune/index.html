💫 日期处理与转换：农历与阳历互换 💫

import streamlit as st
from datetime import datetime
import time

# 日期选择器
selected_date = st.date_input("选择你的生日（阳历）", datetime.now())

# 将日期转换为时间戳
timestamp = time.mktime(selected_date.timetuple())

# 格式化日期显示
formatted_date = selected_date.strftime("%Y年%m月%d日")
st.write(f"你选择的日期是：{formatted_date}")

# 日期相关计算
birth_year = selected_date.year
current_year = datetime.now().year
age = current_year - birth_year
st.write(f"您今年大约 {age} 岁")

# 计算星座（简化版）
month = selected_date.month
day = selected_date.day

zodiac_signs = [
    ("水瓶座", (1, 20), (2, 18)),
    ("双鱼座", (2, 19), (3, 20)),
    ("白羊座", (3, 21), (4, 19)),
    ("金牛座", (4, 20), (5, 20)),
    ("双子座", (5, 21), (6, 21)),
    ("巨蟹座", (6, 22), (7, 22)),
    ("狮子座", (7, 23), (8, 22)),
    ("处女座", (8, 23), (9, 22)),
    ("天秤座", (9, 23), (10, 23)),
    ("天蝎座", (10, 24), (11, 22)),
    ("射手座", (11, 23), (12, 21)),
    ("魔羯座", (12, 22), (1, 19))
]

user_zodiac = "未知"
for sign, (start_month, start_day), (end_month, end_day) in zodiac_signs:
    if (month == start_month and day >= start_day) or (month == end_month and day <= end_day):
        user_zodiac = sign
        break

st.write(f"您的星座是：{user_zodiac}")

    • 自动计算年龄、星座、生肖等信息
    • 实现复杂的时间相关功能

import streamlit as st
from datetime import datetime

st.title('AI生辰八字解析')
st.write('🏮 输入姓名和生日，探索你的命理奥秘！')

# 姓名输入
name = st.text_input('请输入你的姓名', placeholder='例如：张三')

# 创建日期选择器
birth_date = st.date_input('请选择你的出生日期（阳历）', datetime(1990, 1, 1))

# 时辰选择
chinese_hours = ["子时 (23:00-01:00)", "丑时 (01:00-03:00)", "寅时 (03:00-05:00)", 
                "卯时 (05:00-07:00)", "辰时 (07:00-09:00)", "巳时 (09:00-11:00)", 
                "午时 (11:00-13:00)", "未时 (13:00-15:00)", "申时 (15:00-17:00)", 
                "酉时 (17:00-19:00)", "戌时 (19:00-21:00)", "亥时 (21:00-23:00)"]

birth_hour = st.selectbox('请选择你的出生时辰（可选）', ["不清楚"] + chinese_hours)

# 查询按钮
if st.button('开始解析'):
    if not name:
        st.error('请输入姓名')
    else:
        # 格式化生日
        formatted_date = birth_date.strftime("%Y-%m-%d")
        
        # 提取时辰
        if birth_hour != "不清楚":
            chinese_hour = birth_hour.split(" ")[0]
        else:
            chinese_hour = ""
            
        st.write(f"正在解析 {name} 的生辰八字...")

# 时辰选择
chinese_hours = ["子时 (23:00-01:00)", "丑时 (01:00-03:00)", "寅时 (03:00-05:00)", 
                "卯时 (05:00-07:00)", "辰时 (07:00-09:00)", "巳时 (09:00-11:00)", 
                "午时 (11:00-13:00)", "未时 (13:00-15:00)", "申时 (15:00-17:00)", 
                "酉时 (17:00-19:00)", "戌时 (19:00-21:00)", "亥时 (21:00-23:00)"]

birth_hour = st.selectbox('请选择你的出生时辰（可选）', ["不清楚"] + chinese_hours)

# 隐私声明
st.caption("🔒 您的个人信息仅用于生成分析结果，不会被保存或用于其他用途")

# 查询按钮
if st.button('开始解析'):
    if not name:
        st.error('请输入姓名')
    else:
        # 显示加载状态
        with st.spinner('正在解析命理信息...'):
            try:
                # 格式化生日
                formatted_date = birth_date.strftime("%Y-%m-%d")
                
                # 提取时辰
                if birth_hour != "不清楚":
                    chinese_hour = birth_hour.split(" ")[0]
                else:
                    chinese_hour = ""
                
                # 调用API获取命理分析
                fortune_data = lq.ai_fortune(
                    password='lqcode', 
                    name=name, 
                    birthdate=formatted_date, 
                    birthtime=chinese_hour
                )
                
                # 检查是否成功获取数据
                if "name_analysis" in fortune_data:
                    st.success('解析完成！')
                    
                    # 显示姓名解析
                    st.markdown(f"""
                    <div class="chinese-theme">
                        <div class="chinese-title"><span class="chinese-symbol">姓</span> 姓名解析</div>
                        <div class="fortune-section">
                            <div class="fortune-label">姓名寓意：</div>
                            <div class="fortune-content">{fortune_data["name_analysis"]}</div>
                        </div>
                    </div>
                    """, unsafe_allow_html=True)
                    
                    # 显示生辰八字
                    st.markdown(f"""
                    <div class="chinese-theme">
                        <div class="chinese-title"><span class="chinese-symbol">命</span> 生辰八字</div>
                        <div class="fortune-section">
                            <div class="fortune-label">八字：</div>
                            <div class="fortune-content">{fortune_data["bazi"]}</div>
                            
                            <div class="fortune-label">五行分析：</div>
                            <div class="fortune-content">{fortune_data["wuxing"]}</div>
                        </div>
                    </div>
                    """, unsafe_allow_html=True)

                        <div class="chinese-title"><span class="chinese-symbol">相</span> 性格特点</div>
