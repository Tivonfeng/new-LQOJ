🧰 3.2认识新的编程概念

💫 计时器与状态管理：让游戏更刺激 💫

import streamlit as st
import time
from datetime import datetime, timedelta

# 设置初始状态
if 'start_time' not in st.session_state:
    st.session_state.start_time = None
if 'remaining_time' not in st.session_state:
    st.session_state.remaining_time = 60  # 60秒倒计时

# 开始游戏按钮
if st.button('开始游戏') and not st.session_state.start_time:
    st.session_state.start_time = datetime.now()

# 显示倒计时
if st.session_state.start_time:
    elapsed = (datetime.now() - st.session_state.start_time).total_seconds()
    remaining = max(0, st.session_state.remaining_time - int(elapsed))
    
    # 创建进度条
    progress = 1 - (remaining / st.session_state.remaining_time)
    st.progress(progress)
    
    # 显示剩余时间
    st.write(f"剩余时间：{remaining} 秒")
    
    # 检查游戏是否结束
    if remaining <= 0:
        st.error("时间到！游戏结束！")
        st.session_state.start_time = None  # 重置游戏
  • 🖥️ 计时器与游戏状态的优势：
    • 创造紧张刺激的游戏氛围
    • 记录用户的游戏进程和成绩
    • 实现游戏的开始、暂停和结束功能

😎 4.1 小试牛刀｜创建计时器

import streamlit as st
import time
from datetime import datetime

st.title('知识竞答游戏')
st.write('⏱️ 一分钟内答对尽可能多的题目！')

# 设置初始状态
if 'start_time' not in st.session_state:
    st.session_state.start_time = None
if 'game_active' not in st.session_state:
    st.session_state.game_active = False

# 开始游戏按钮
if st.button('开始游戏'):
    st.session_state.start_time = datetime.now()
    st.session_state.game_active = True
    st.experimental_rerun()

# 显示倒计时
if st.session_state.game_active:
    # 创建占位符
    timer_placeholder = st.empty()
    
    # 倒计时60秒
    total_time = 60
    start_time = st.session_state.start_time
    
    # 实时更新倒计时
    while st.session_state.game_active:
        now = datetime.now()
        elapsed = (now - start_time).total_seconds()
        remaining = max(0, total_time - int(elapsed))
        
        # 更新进度条和时间
        progress = 1 - (remaining / total_time)
        timer_placeholder.progress(progress)
        timer_placeholder.write(f"⏱️ 剩余时间：{remaining} 秒")
        
        # 检查游戏是否结束
        if remaining <= 0:
            st.session_state.game_active = False
            st.error("⏰ 时间到！游戏结束！")
            st.balloons()
            break
            
        # 暂停一小段时间，避免过度刷新
        time.sleep(0.1)

from lqcodeAI import lq
import streamlit as st
import time
from datetime import datetime
import random

st.title('AI知识竞答游戏')
st.write('⏱️ 一分钟内答对尽可能多的题目！')

# 初始化会话状态
if 'questions' not in st.session_state:
    st.session_state.questions = []
if 'current_question' not in st.session_state:
    st.session_state.current_question = 0
if 'score' not in st.session_state:
    st.session_state.score = 0
if 'answered' not in st.session_state:
    st.session_state.answered = 0
if 'start_time' not in st.session_state:
    st.session_state.start_time = None
if 'game_active' not in st.session_state:
    st.session_state.game_active = False

# 选择题目领域
topic_options = ["科技", "历史", "地理", "体育", "娱乐", "艺术", "科学", "文学"]
selected_topic = st.selectbox(
    '选择题目领域',
    topic_options
)

# 开始游戏按钮
if st.button('开始游戏'):
    with st.spinner('AI正在出题中...'):
        # 获取题目
        questions = lq.ai_quiz(password='lqcode', topic=selected_topic, count=20)
        
        # 随机打乱题目顺序
        random.shuffle(questions)
        
        # 保存题目和重置状态
        st.session_state.questions = questions
        st.session_state.current_question = 0
        st.session_state.score = 0
        st.session_state.answered = 0
        st.session_state.start_time = datetime.now()
        st.session_state.game_active = True
        
    st.success('准备好了！开始答题吧！')
    st.experimental_rerun()

# 游戏进行中
if st.session_state.game_active and st.session_state.questions:
    # 创建倒计时占位符
    timer_col1, timer_col2 = st.columns([3, 1])
    progress_placeholder = timer_col1.empty()
    time_placeholder = timer_col2.empty()
    
    # 显示分数
    score_placeholder = st.empty()
    score_placeholder.write(f"💯 当前得分: {st.session_state.score} | 已答题数: {st.session_state.answered}")
    
    # 显示当前题目
    question_data = st.session_state.questions[st.session_state.current_question]
    question_placeholder = st.empty()
    question_placeholder.write(f"**问题 {st.session_state.current_question + 1}**: {question_data['question']}")
    
    # 选项按钮
    option_columns = st.columns(4)
    options = question_data['options']
    correct_answer = question_data['answer']
    
    # 更新倒计时
    total_time = 60
    start_time = st.session_state.start_time
    now = datetime.now()
    elapsed = (now - start_time).total_seconds()
    remaining = max(0, total_time - int(elapsed))
    
    # 显示进度条和时间
    progress = 1 - (remaining / total_time)
    progress_placeholder.progress(progress)
    time_placeholder.write(f"⏱️ {remaining}秒")
    
    # 检查游戏是否结束
    if remaining <= 0:
        st.session_state.game_active = False
        st.error("⏰ 时间到！游戏结束！")
        
        # 显示最终成绩
        st.success(f"🎮 游戏结束！")
        st.write(f"💯 最终得分: {st.session_state.score}")
        st.write(f"📝 答题数量: {st.session_state.answered}")
        st.write(f"✅ 正确率: {int(st.session_state.score / max(1, st.session_state.answered) * 100)}%")
        
        # 根据得分显示评价
        if st.session_state.score >= 15:
            st.write("🌟 知识达人！太厉害了！")
        elif st.session_state.score >= 10:
            st.write("👍 表现不错！继续加油！")
        elif st.session_state.score >= 5:
            st.write("💪 再接再厉！你可以做得更好！")
        else:
            st.write("🔍 多学习，下次一定会更好！")
            
        st.balloons()
    
    # 显示选项按钮
    else:
        # 处理答题
        def answer_question(selected_option, correct_option):
            # 增加已答题数
            st.session_state.answered += 1
            
            # 检查答案
            if selected_option == correct_option:
                st.session_state.score += 1
                st.success("✅ 回答正确！")
            else:
                st.error(f"❌ 回答错误！正确答案是: {correct_option}")
            
            # 移动到下一题
            st.session_state.current_question = (st.session_state.current_question + 1) % len(st.session_state.questions)
            time.sleep(0.5)  # 短暂停留显示结果
            st.experimental_rerun()
        
        # 创建选项按钮
        option_labels = ['A', 'B', 'C', 'D']
        for i, (col, option, label) in enumerate(zip(option_columns, options, option_labels)):
            col.button(f"{label}. {option}", key=f"option_{i}", on_click=answer_question, args=(label, correct_answer))
else:
    if st.session_state.start_time:  # 游戏已结束
        # 显示重新开始按钮
        if st.button('再玩一次'):
            st.session_state.start_time = None
            st.experimental_rerun()
