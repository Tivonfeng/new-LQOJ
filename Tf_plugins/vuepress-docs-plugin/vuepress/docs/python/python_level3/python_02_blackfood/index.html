💫 字典（Dictionary）：存储数据的魔法袋 💫

# 字典是键值对的集合
gadget = {
    "name": "时光记忆棒棒糖",
    "ingredients": ["蓝莓", "时间胶囊", "魔法糖浆"],
    "effect": "回忆童年记忆"
}

# 访问字典元素
print(gadget["name"])  # 输出: 时光记忆棒棒糖
  • 🖥️ 字典的优势：
    • 可以存储不同类型的数据
    • 通过键快速访问值
    • 非常适合表示复杂的结构化数据

from lqcodeAI import lq
import streamlit as st

st.title('哆啦A梦零食道具创作大师')
st.write('🍳 输入你的食材，AI帮你创造神奇道具！')

preferences = st.text_area('请输入食材（用逗号分隔）', placeholder='例如：棉花糖,彩虹糖,时间粉,记忆豆')

if st.button('✨ 开始创作'):
    if preferences:
        with st.spinner('哆啦A梦正在四次元口袋里翻找材料...'):
            result = lq.ai_snackprops(password='lqcode', preferences=preferences)
            
        st.success('道具创作完成！')
        
        # 使用分栏布局展示结果
        col1, col2 = st.columns(2)
        with col1:
            st.subheader('🍽️ ' + result['titile'])
            st.write(result['appearance'])
        with col2:
            st.subheader('✨ 神奇功效')
            st.write(result['function'])
            
        # 添加副作用和警告信息
        st.subheader('⚠️ 副作用')
        st.write(result['sub_fuction'])
        
        st.subheader('🚨 警告')
        st.write(result['warning'])
            
        st.balloons()
    else:
        st.error('请输入至少一种食材哦！')
