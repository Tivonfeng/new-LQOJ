🛠️ 二、知识讲解（20分钟）

🧰 2.1 算术运算符

✨ 基本运算 ✨

# 加减乘除
print(1 + 2)    # 加法：3
print(5 - 3)    # 减法：2
print(4 * 2)    # 乘法：8
print(10 / 2)   # 除法：5.0

# 取余和整除
print(10 % 3)   # 取余：1
print(10 // 3)  # 整除：3

# 幂运算
print(2 ** 3)   # 2的3次方：8

🧰 2.2 比较运算符

💫 比较运算 💫

# 等于和不等于
print(1 == 1)   # 等于：True
print(1 != 2)   # 不等于：True

# 大于和小于
print(5 > 3)    # 大于：True
print(2 < 4)    # 小于：True

# 大于等于和小于等于
print(5 >= 5)   # 大于等于：True
print(3 <= 4)   # 小于等于：True

🧰 2.3 逻辑运算符

and运算
# and：两个条件都为True才返回True
print(True and True)    # True
print(True and False)   # False
print(False and False)  # False
or运算
# or：只要有一个条件为True就返回True
print(True or True)     # True
print(True or False)    # True
print(False or False)   # False
not运算
# not：取反
print(not True)         # False
print(not False)        # True

🧰 2.4 赋值运算符

基本赋值
# 基本赋值
a = 1
b = 2
print(a, b)  # 1 2
复合赋值
# 复合赋值
a = 1
a += 2  # 等价于 a = a + 2
print(a)  # 3

b = 5
b *= 2  # 等价于 b = b * 2
print(b)  # 10

💖 3.2 进阶挑战

# 挑战：计算BMI指数
height = float(input("请输入你的身高(m)："))
weight = float(input("请输入你的体重(kg)："))
bmi = weight / (height ** 2)
print(f"你的BMI指数是：{bmi:.2f}")

if bmi < 18.5:
    print("体重偏轻")
elif bmi < 24:
    print("体重正常")
else:
    print("体重偏重")
